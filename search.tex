	\end{definition}

\frame
{
	\frametitle{Linear Search}
	
	\begin{algorithmic}
		\Procedure{Linear Search}{$*begin, *end, target$}
			\State$result\gets\infty$
            \For{$itr\in[begin, end)$}
            	\If{$*itr\geq target\land *itr < result$}
					\State$result\gets *itr$
				\EndIf
            \EndFor
            \State\Return$result$
        \EndProcedure
	\end{algorithmic}
}
